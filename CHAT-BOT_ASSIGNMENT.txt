# Import necessary libraries
from transformers import pipeline

# Define the model ID
model_id = 'google/flan-t5-base'

# Load the chatbot pipeline
chatbot = pipeline('text2text-generation', model=model_id)

# Define the chat function
def chat(user_input, user_role):
    prompt = f'As a {user_role}, please answer the following: {user_input}'
    reply = chatbot(prompt, max_new_tokens=100)[0]['generated_text']
    return reply

# Start the chat loop
print('Hii Surya...its your BOT..let"s talk\nif you dont want press exit..')
while True:
    inp = input('You:')
    if inp.lower() == 'exit':
        print('OK....byy..\nThank You')
        break
    else:
        role = input("Enter your role:-")
        reply = chat(inp, role)
        print('Bot:', reply)
